Example 1. Simple one 

   public static synchronized void main(String args[])
   {
     try {
       // Create a default parameter validator
       ParameterValidator pv = new ParameterValidator();
       // Create an empty command set
       Set<Command> cs = new HashSet();
       // Create the interpreter
       NaturalCLI nc = new NaturalCLI(cs, pv);
       // Add the commands that can be understood
       cs.add(new HelpCommand(cs));
       cs.add(new HTMLHelpCommand(cs));
       cs.add(new SleepCommand());  
       cs.add(new ExecuteFileCommand(nc));
       // Execute the command lines  
       nc.execute(args, 0);
     } catch (Exception e) {
       e.printStackTrace();
     }
   }
